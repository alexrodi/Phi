189 void Connections::patchCordTypeBCallback (Path& path, Point<float> positionA, Point<float> positionB)
190 {

192

194 
195  bool order = positionA.x > positionB.x;

197  Point<float> cubicHandleA = positionA.translated(order ? -hDistance : hDistance, 0);

199 
200  path.clear();

202  path.cubicTo (positionA , cubicHandleA , middlePoint);
