43 static void drawWaveform(juce::Graphics &g, const Rectangle<float> &viewPort, float shape, float decay, Colour colour) {
44 
45  g.setColour(colour);
46 
47  const int pixelsPerPoint = 2;
48  const float strokeWidth = 2;
49 
50  const float startX = viewPort.getX();
51  const float centreY = viewPort.getCentreY();
52  const float yRange = (viewPort.getHeight() * 0.5) - 3;
53  const float width = viewPort.getWidth();
54  const float endX = startX+width;
55  const float shapeValue = 1.006 - shape;
56  const int aaValue = 8; // x8 AA
57  const float phaseIncrement = ((7/width)*pixelsPerPoint)/aaValue; // go up to x=7 (arbitrary value)
58 
59  float phase = 0;
60 
61  // Waveform path
62  Path wavePath;
63  wavePath.startNewSubPath (startX, centreY);
64 
65  // Add lines to path
66  for (int x=startX; x<endX; x += pixelsPerPoint){
67  float y = 0;
68  for(int i=0; i<aaValue; i++)
69  {
70  const float lengthPhase = phase*decay;
71  // This function is not safe for audio routines! it has a singularity at x=PI :)
72  const float newY = (lengthPhase==PI) ? 0.f : sin( sin(lengthPhase) / (shapeValue * (lengthPhase-PI)) );
73  y += abs(newY);
74  phase += phaseIncrement;
75  }
76  y /= aaValue;
77  wavePath.lineTo (x, y * yRange + centreY);
78  }
79 
80  wavePath.lineTo (endX, centreY);
81 
82  wavePath = wavePath.createPathWithRoundedCorners(60);
83 
84  g.setGradientFill(ColourGradient().vertical(colour, centreY+yRange, colour.darker(), centreY));
85  g.fillPath(wavePath);
86  g.strokePath (wavePath, PathStrokeType (2.0f));
87 
88  g.setGradientFill(ColourGradient().vertical(colour, centreY-yRange, colour, centreY));
89  wavePath.applyTransform(AffineTransform().verticalFlip(centreY+yRange+strokeWidth*2));
90  g.fillPath(wavePath);
91  g.strokePath (wavePath, PathStrokeType (strokeWidth));
92 }
